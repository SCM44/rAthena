prt_in,38,103,4	script	Platinum Skill NPC	94,{
	mes "[Platinum Skill NPC]";
	mes "Je peux te donner les skills spéciaux pour ta classe.";
	mes "Tu les veux ?";
	next;
	if (select("Oui s'il te plait:Non") == 2) {
		mes "[Platinum Skill NPC]";
		mes "Passes une bonne journée... >.>";
		close;
	}
	callfunc "F_GetPlatinumSkills";
	mes "[Platinum Skill NPC]";
	mes "Et voilà !";
	close;
}
